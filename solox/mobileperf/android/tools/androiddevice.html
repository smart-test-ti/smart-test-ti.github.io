  56    def get_adb_path():
  57        '''返回adb.exe的绝对路径。优先使用指定的adb，若环境变量未指定，则返回当前脚本tools目录下的adb
  58
  59        :return: 返回adb.exe的绝对路径
  60        :rtype: str
  61        '''
  62        if ADB.adb_path:
  63            return ADB.adb_path
  64        ADB.adb_path = os.environ.get('ADB_PATH')
  65        if ADB.adb_path != None and ADB.adb_path.endswith('adb.exe'):
  66            return ADB.adb_path
  67        # 判断系统默认adb是否可用，如果系统有配，默认优先用系统的，避免5037端口冲突
  68        proc = subprocess.Popen('adb devices', stdout=subprocess.PIPE, shell=True)
  69        result = proc.stdout.read()
  70        # logger.debug(result)
  71        if not isinstance(result, str):
  72            result = str(result, 'utf-8')
  73        # 说明自带adb  windows上返回结果不是这样 另外有可能第一次执行，adb会不正常
  74        if result and "command not found" not in result:
  75            ADB.adb_path = "adb"
  76            # logger.debug("system have adb")
  77            return ADB.adb_path
  78        # logger.debug("system have no adb")
  79        cur_path = os.path.dirname(os.path.abspath(__file__))
  80        ADB.os_name = platform.system()
  81        # logger.debug("platform :" + ADB.os_name)
  82        if ADB.os_name == "Windows":
  83            ADB.adb_path = os.path.join(cur_path, u'adb.exe')
  84        elif ADB.os_name == "Darwin":
  85            ADB.adb_path = os.path.join(cur_path, "platform-tools-latest-darwin", "platform-tools", "adb")
  86        else:

 108    def list_device():
 109        '''获取设备列表
 110
 111        :return: 返回设备列表
 112        :rtype: list
 113        '''
 114        proc = subprocess.Popen("adb devices", stdout=subprocess.PIPE, shell=True)
 115        result = proc.stdout.read()
 116        if not isinstance(result, str):
 117            result = result.decode('utf-8')
 118        result = result.replace('\r', '').splitlines()
 119        # logger.debug("adb devices:")
 120        # logger.debug(result)
 121        device_list = []
 122        for device in result[1:]:
 123            if len(device) <= 1 or not '\t' in device: continue
 124            if device.split('\t')[1] == 'device':
 125                # 只获取连接正常的
 126                device_list.append(device.split('\t')[0])
 127        return device_list

 196    def _timer(self, process, timeout):
 197        '''进程超时器，监控adb同步命令执行是否超时，超时强制结束执行。当timeout<=0时，永不超时
 198
 199        :param Popen process: 子进程对象
 200        :param int timeout: 超时时间
 201        '''
 202        num = 0
 203        while process.poll() == None and num < timeout * 10:
 204            num += 1
 205            time.sleep(0.1)
 206        if process.poll() == None:
 207            # logger.warning("%d process timeout,force close" % process.pid)
 208            process.terminate()

 210    def _run_cmd_once(self, cmd, *argv, **kwds):
 211        '''执行一次adb命令：cmd
 212
 213        :param str cmd: 命令字符串
 214        :param list argv: 可变参数
 215        :param dict kwds: 可选关键字参数 (超时/异步)
 216        :return: 执行adb命令的子进程或执行的结果
 217        :rtype: Popen or str
 218        '''
 219        import locale
 220        import codecs
 221        if self._device_id:
 222            cmdlet = [self._adb_path, '-s', self._device_id, cmd]
 223        else:
 224            cmdlet = [self._adb_path, cmd]
 225        for i in range(len(argv)):
 226            arg = argv[i]
 227            if not isinstance(argv[i], str):
 228                arg = arg.decode('utf8')
 229            cmdlet.append(arg)
 230        #        logger.debug('ADB cmd:' + " ".join(cmdlet))
 231        #         logger.debug(cmdlet)
 232        cmdStr = " ".join(cmdlet)
 233        # logger.debug(cmdStr)
 234        process = None
 235        #       windows上 不要传cmdStr 目录有空格，会报错
 236        #         if ADB.os_name == "Windows":
 237        #             process = subprocess.Popen(cmdlet, stdin=subprocess.PIPE, stdout=subprocess.PIPE, stderr=subprocess.PIPE,shell=True)
 238        # #       mac上传list会报错:Android Debug Bridge version
 239        #         else:
 240        #         windows ["adb devices"] 提示没有命令 ，改为str执行
 241        process = subprocess.Popen(cmdStr, stdin=subprocess.PIPE, stdout=subprocess.PIPE, stderr=subprocess.PIPE,
 242                                   shell=True)
 243        if "sync" in kwds and kwds['sync'] == False:
 244            # 异步执行命令，不等待结果，返回该子进程对象
 245            return process
 246        before = time.time()
 247        timeout = 10
 248        if "timeout" in kwds:
 249            timeout = kwds['timeout']
 250        if timeout != None and timeout > 0:
 251            # timeout = None 或者小于等于0时，一直等待执行结果
 252            threading.Thread(None, self._timer, (process, timeout))
 253        (out, error) = process.communicate()
 254        # 执行错误 mac  out无输出 error有输出 返回值非0
 255        # 执行错误 windows out有输出 error没有输出，返回值0
 256        if process.poll() != 0:  # 返回码为非0，表示命令未执行成功返回
 257            # logger.logError("adb执行出错或超时，出错命令是：\n%s"%cmdlet)
 258            if error and len(error) != 0:
 259                logger.debug("adb error info:\n%s" % error)
 260            if "no devices/emulators found" in str(out) or "no devices/emulators found" in str(error):
 261                logger.error("no devices/emulators found,please reconnect phone,make sure adb shell normal")
 262                return ""
 263            #               退出整个进程
 264            if "killing" in str(out) or "killing" in str(error):
 265                logger.error(
 266                    "adb 5037 port is occupied,please stop the process occupied 5037 port,make sure adb devices normal")
 267                return ""
 268            if "device not found" in str(out) or "device not found" in str(error):
 269                logger.error("device not found,please reconnect phone,make sure adb devices normal")
 270                self.before_connect = False
 271                self.after_connect = False
 272                return ""
 273            if "offline" in str(out) or "offline" in str(error):
 274                logger.error("device offline,please reconnect phone,make sure adb devices normal")
 275                return ""
 276            if "more than one" in str(out) or "more than one" in str(error):
 277                logger.error("more than one device,please input device serialnum!")
 278                # sys.exit(1)
 279            if "Android Debug Bridge version" in str(out) or "Android Debug Bridge version" in str(error):
 280                logger.error("adb cmd error!:" + out)
 281                # sys.exit(1)
 282        if str(out, "utf-8") == '':
 283            # logger.debug("out is empty ,use error")
 284            out = error
 285        self.after_connect = True
 286        after = time.time()
 287        time_consume = after - before
 288        # logger.info(cmdStr + " time consume: " + str(time_consume))
 289        if not isinstance(out, str):
 290            try:
 291                out = str(out, "utf8")
 292            except Exception as e:
 293                out = repr(out)
 294        return out.strip()

 296    def run_adb_cmd(self, cmd, *argv, **kwds):
 297        '''尝试执行adb命令
 298
 299        :param str cmd: 命令字符串
 300        :param list argv: 可变参数
 301        :param dict kwds: 可选关键字参数 (超时/异步)
 302        :return: 执行adb命令的子进程或执行的结果
 303        :rtype: Popen or str
 304        '''
 305        retry_count = 3  # 默认最多重试3次
 306        if "retry_count" in kwds:
 307            retry_count = kwds['retry_count']
 308        while retry_count > 0:
 309            ret = self._run_cmd_once(cmd, *argv, **kwds)
 310            if ret != None:
 311                break
 312            retry_count = retry_count - 1
 313        return ret

 410    def start_logcat(self, save_dir, process_list=[], params=''):
 411        '''运行logcat进程
 412
 413        :param list process_list: 要捕获日志的进程名或进程ID列表，为空则捕获所有进程,输入 ['system_server']可捕获系统进程的日志
 414        :param str params: 参数

 417            os.makedirs(save_dir)

 986    def list_process(self):
 987        '''获取进程列表
 988        '''
 989        # <= 7.0 用ps, >=8.0 用ps -A android8.0 api level 26
 990        result = None
 991        if self.get_sdk_version() < 26:
 992            result = self.run_shell_cmd('ps')  # 不能使用grep
 993        else:
 994            result = self.run_shell_cmd('ps -A')  # 不能使用grep
 995        result = result.replace('\r', '')
 996        lines = result.split('\n')
 997        busybox = False
 998        if lines[0].startswith('PID'): busybox = True
 999
1000        result_list = []
1001        for i in range(1, len(lines)):
1002            items = lines[i].split()
1003            if not busybox:
1004                if len(items) < 9:
1005                    err_msg = "ps命令返回格式错误：\n%s" % lines[i]
1006                    if len(items) == 8:
1007                        result_list.append({'uid': items[0], 'pid': int(items[1]), 'ppid': int(items[2]),
1008                                            'proc_name': items[7], 'status': items[-2]})
1009                    else:
1010                        logger.error(err_msg)
1011                else:
1012                    result_list.append({'uid': items[0], 'pid': int(items[1]), 'ppid': int(items[2]),
1013                                        'proc_name': items[8], 'status': items[-2]})
1014            else:
1015                idx = 4
1016                cmd = items[idx]
1017                if len(cmd) == 1:
1018                    # 有时候发现此处会有“N”
1019                    idx += 1
1020                    cmd = items[idx]
1021                idx += 1
1022                if cmd[0] == '{' and cmd[-1] == '}': cmd = items[idx]
1023                ppid = 0
1024                if items[1].isdigit(): ppid = int(items[1])  # 有些版本中没有ppid
1025                result_list.append({'pid': int(items[0]), 'uid': items[1], 'ppid': ppid,
1026                                    'proc_name': cmd, 'status': items[-2]})
1027        return result_list

 298        '''尝试执行adb命令

 411    def start_logcat(self, save_dir, process_list=[], params=''):
 412        '''运行logcat进程
 413
 414        :param list process_list: 要捕获日志的进程名或进程ID列表，为空则捕获所有进程,输入 ['system_server']可捕获系统进程的日志
 415        :param str params: 参数
 416        '''
 417        if not os.path.exists(save_dir):
 418            os.makedirs(save_dir)
 419        if hasattr(self, '_logcat_running') and self._logcat_running == True:
 420            logger.warning('logcat process have started,not need start')

 573            返回文件绝对路径 列表
