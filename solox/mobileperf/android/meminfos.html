 29class MemInfoPackage(object):
 30    RE_PROCESS = re.compile(r'\*\* MEMINFO in pid (\d+) \[(\S+)] \*\*')
 31    RE_TOTAL_PSS = re.compile(r'TOTAL\s+(\d+)')
 32    RE_JAVA_HEAP = re.compile(r"Java Heap:\s+(\d+)")
 33    RE_Native_HEAP = re.compile(r"Native Heap:\s+(\d+)")
 34    RE_System = re.compile(r"System:\s+(\d+)")
 35
 36    pid = 0
 37    processName = ''
 38    datetime = ''
 39    totalPSS = 0
 40    totalAllocHeap = 0
 41    javaHeap = 0
 42    nativeHeap = 0
 43    system = 0
 44
 45    def __init__(self,  dump):
 46        self.dump = dump
 47        self._parse()

 50    def _parse(self):
 51        '''
 52        dumpsys meminfo package 中解析出需要的数据，由于版本变迁，这个数据的结构变化较多，比较了不同版本发现这两列数据total pss和Heap Alloc是都有的，而且这两个指标对于展示
 53        应用性能指标还是比较有代表性的。
 54        :return:
 55        '''
 56        match = self.RE_PROCESS.search(self.dump)
 57        if match:
 58            self.pid = match.group(1)
 59            self.processName = match.group(2)
 60        match = self.RE_TOTAL_PSS.search(self.dump)
 61        if match:
 62            self.totalPSS = round(float(match.group(1))/1024,2)
 63
 64        match = self.RE_JAVA_HEAP.search(self.dump)
 65        if match:
 66            self.javaHeap = round(float(match.group(1)) / 1024, 2)
 67
 68        match = self.RE_Native_HEAP.search(self.dump)
 69        if match:
 70            self.nativeHeap = round(float(match.group(1)) / 1024, 2)
 71
 72        match = self.RE_System.search(self.dump)
 73        if match:
 74            self.system = round(float(match.group(1)) / 1024, 2)
 75
 76        result = self.dump.split('\n')#需要将其转为列表
 77
 78        for line in result:
 79            if "TOTAL" in line and ":" not in line:
 80                tmp = line.split()
 81                self.totalAllocHeap = round(float(tmp[-2])/1024,2)

103    def _parse(self):
104        '''
105        从dumpsys meminfo中解析出Total RAM，Free RAM, 和Used RAM这几个值并保存在相关实例变量中
106        :return: NONE
107        '''
108        # logger.debug(self.dump)
109        match = self.RE_TOTAL_MEMORY.search(self.dump)
110        if match:
111            self.totalmem = round(float(match.group(1).replace(",",""))/1024,2)
112        match = self.RE_FREE_MEMORY.search(self.dump)
113        if match:
114            self.freemem = round(float(match.group(1).replace(",",""))/1024,2)
115        match = self.RE_USED_MEMORY.search(self.dump)
116        if match:
117            self.usedmem = round(float(match.group(1).replace(",",""))/1024,2)
118
119        logger.debug(" device general mem，total mem: "+str(self.totalmem) + ", \n used mem: " + str(self.usedmem) + ", free mem: " + str(self.freemem))
120        for package in self.packages:
121            # 可能子进程没有启动，默认填空值 方便格式上统一处理
122            mem_dic = {"package": package, "pid": "", "pss": ""}
123            RE_PROCESS_MEMORY = re.compile(r"([\d,]+)\s*(K|kB):\s+" + package + "\s+\(pid\s+(\d+)")
124            # 252,370K: com.alibaba.ailabs.tg (pid 26620 / activities)
125            # 111920 kB: com.alibaba.ailabs.tg (pid 16036 / activities)
126            RE_PROCESS_MEMORY_2 = re.compile(r"([\d,]+)\s+kB:\s+\d+\s+kB:\s+" + package + "\s+\(pid\s+(\d+)")
127            # 243786 kB:       0 kB: com.alibaba.ailabs.tg (pid 16993 / activities)
128            for line in self.dump.splitlines():
129                match = RE_PROCESS_MEMORY.search(line)
130                match2 = RE_PROCESS_MEMORY_2.search(line)
131                if match:
132                    pss = round(float(match.group(1).replace(",", "")) / 1024, 2)
133                    mem_dic = {"package": package, "pid": match.group(3), "pss": str(pss)}
134                    # self.package_pid_pss_list.append(mem_dic)
135                    # logger.debug("line:"+line +" package:"+package+" pid:"+match.group(3)+" pss:"+str(pss))
136                    self.total_pss = self.total_pss + pss
137                    break
138                elif match2:
139                    pss = round(float(match2.group(1).replace(",", "")) / 1024, 2)
140                    mem_dic = {"package": package, "pid": match2.group(2), "pss": str(pss)}
141                    # self.package_pid_pss_list.append(mem_dic)
142                    # logger.debug("line:"+line +" package:"+package+" pid:"+match2.group(2)+" pss:"+str(pss))
143                    self.total_pss = self.total_pss + pss
144                    break
145            self.package_pid_pss_list.append(mem_dic)
146            logger.debug(mem_dic)

220    def _collect_memory_thread(self, start_time):
221        end_time = time.time() + self._timeout
222        mem_list_titile = ["datatime", "total_ram(MB)", "free_ram(MB)"]
223        pid_list_titile = ["datatime"]
224        pss_detail_titile = ["datatime","package","pid","pss","java_heap","native_heap","system"]
225        for i in range(0,len(self.packages)):
226            mem_list_titile.extend(["package", "pid", "pid_pss(MB)"])
227            pid_list_titile.extend(["package", "pid"])
228        if len(self.packages)>1:
229            mem_list_titile.append("total_pss(MB)")
230        mem_file = os.path.join(RuntimeData.package_save_path, 'meminfo.csv')
231        pid_file = os.path.join(RuntimeData.package_save_path, 'pid_change.csv')
232        for package in self.packages:
233            pss_detail_file = os.path.join(RuntimeData.package_save_path, 'pss_%s.csv'%package.split(".")[-1].replace(":","_"))
234            with open(pss_detail_file, 'a+',encoding="utf-8") as df:
235                csv.writer(df, lineterminator='\n').writerow(pss_detail_titile)
236        try:
237            with open(mem_file, 'a+',encoding="utf-8") as df:
238                csv.writer(df, lineterminator='\n').writerow(mem_list_titile)
239                if self.mem_queue:
240                    mem_file_dic = {'mem_file': mem_file}
241                    self.mem_queue.put(mem_file_dic)
242
243            with open(pid_file, 'a+',encoding="utf-8") as df:
244                csv.writer(df, lineterminator='\n').writerow(pid_list_titile)
245        except RuntimeError as e:
246            logger.error(e)
247        starttime_stamp = TimeUtils.getTimeStamp(start_time,"%Y_%m_%d_%H_%M_%S")
248        old_package_pid_pss_list = []
249        dumpsys_mem_times = 0
250        # D系统上会报错 System server has no access to file context
251        # hprof_path = "/sdcard/hprof"
252        hprof_path = "/data/local/tmp"
253        self.device.adb.run_shell_cmd("mkdir "+hprof_path)
254        # sdcard 卡目录下dump需要打开这个开关
255        self.device.adb.run_shell_cmd("setenforce 0")
256        first_dump = True
257        while not self._stop_event.is_set() and time.time() < end_time:
258            try:
259                before = time.time()
260                logger.debug("-----------into _collect_mem_thread loop, thread is : " + str(threading.current_thread().name))
261                collection_time = time.time()
262                # # 获取主进程的详细信息
263                for package in self.packages:
264                    mem_pck_snapshot = self._dumpsys_process_meminfo(package)
265                    if 0 == mem_pck_snapshot.totalPSS:
266                        logger.error("package total pss is 0:%s"%package)
267                        continue
268                    pss_detail_file = os.path.join(RuntimeData.package_save_path,'pss_%s.csv' % package.split(".")[-1].replace(":","_"))
269                    pss_detail_list= [TimeUtils.formatTimeStamp(collection_time),package,mem_pck_snapshot.pid,mem_pck_snapshot.totalPSS,
270                                      mem_pck_snapshot.javaHeap,mem_pck_snapshot.nativeHeap,mem_pck_snapshot.system]
271                    with open(pss_detail_file, 'a+',encoding="utf-8") as pss_writer:
272                        writer_p = csv.writer(pss_writer, lineterminator='\n')
273                        writer_p.writerow(pss_detail_list)
274                #         写到pss_detail表格中
275
276                # 每隔dumpheap_freq分钟， dumpheap一次
277                if (before - starttime_stamp) > RuntimeData.config_dic["dumpheap_freq"] or first_dump:
278                #     先清理hprof文件
279                    filelist = self.device.adb.list_dir(hprof_path)
280                    if filelist:
281                        for file in filelist:
282                            for package in self.packages:
283                                if package in file:
284                                    self.device.adb.delete_file(hprof_path+"/" + file)
285                # if (before - starttime_stamp) % 60 < self._interval and "D" in self.device.adb.get_system_version():
286                    for package in self.packages:
287                        self.device.adb.dumpheap(package,RuntimeData.package_save_path)
288                    starttime_stamp = before
289                    # self.device.adb.run_shell_cmd("kill -10 %s"%str(mem_pck_snapshot.pid))
290                # dumpsys meminfo 耗时长，可能会导致system server cpu占用变高，降低采集频率
291                dumpsys_mem_times = dumpsys_mem_times + 1
292                # 10倍率frequency dumpsys meminfo一次
293                if dumpsys_mem_times%10==0 or first_dump:
294                    mem_device_snapshot = self._dumpsys_meminfo()
295                    # 如果没有采集到dumpsys meminfo的信息，正常情况totalmem不可能为0
296                    if mem_device_snapshot == None or not mem_device_snapshot.package_pid_pss_list or mem_device_snapshot.totalmem == 0:
297                        logger.error("mem_device_snapshot is none")
298                        # 如果获取不到结果，继续延长采集间隔
299                        dumpsys_mem_times = dumpsys_mem_times-1
300                        continue
301                    first_dump = False
302                    logger.debug("current time: " + TimeUtils.getCurrentTime() + ", processname: " + ",total pss:" + str(mem_device_snapshot.total_pss))
303                    logger.debug("collection time in meminfo is : " + TimeUtils.getCurrentTime())
304                    gather_list = [TimeUtils.formatTimeStamp(collection_time), mem_device_snapshot.totalmem, mem_device_snapshot.freemem]
305                    pid_list = [TimeUtils.formatTimeStamp(collection_time)]
306                    pid_change = False
307                    for i in range(0,len(self.packages)):
308                        if len(mem_device_snapshot.package_pid_pss_list) == len(self.packages):
309                            gather_list.extend([mem_device_snapshot.package_pid_pss_list[i]["package"],mem_device_snapshot.package_pid_pss_list[i]["pid"],
310                                           mem_device_snapshot.package_pid_pss_list[i]["pss"]])
311                    if not old_package_pid_pss_list:
312                        old_package_pid_pss_list = mem_device_snapshot.package_pid_pss_list
313                        pid_change = True
314                    else:
315                        for i in range(0, len(self.packages)):
316                            package = mem_device_snapshot.package_pid_pss_list[i]["package"]
317                            if mem_device_snapshot.package_pid_pss_list[i]["pid"] and \
318                                    old_package_pid_pss_list[i]["pid"]!=mem_device_snapshot.package_pid_pss_list[i]["pid"]:
319                                pid_change = True
320                                # 确保上次pid也有
321                                if old_package_pid_pss_list[i]["pid"]:
322                                    if package and package in RuntimeData.config_dic["pid_change_focus_package"]:
323                                        # 确保有tombstones文件才提单
324                                        self.device.adb.pull_file("/data/vendor/tombstones",
325                                                                  RuntimeData.package_save_path)
326                    if pid_change:
327                        old_package_pid_pss_list = mem_device_snapshot.package_pid_pss_list
328                        for i in range(0, len(self.packages)):
329                            if len(old_package_pid_pss_list) == len(self.packages):
330                                pid_list.extend([old_package_pid_pss_list[i]["package"],old_package_pid_pss_list[i]["pid"]])
331                        try:
332                            with open(pid_file, 'a+',encoding="utf-8") as pid_writer:
333                                writer_p = csv.writer(pid_writer, lineterminator='\n')
334                                writer_p.writerow(pid_list)
335                                logger.debug("write to file:" + pid_file)
336                                logger.debug(pid_list)
337                        except RuntimeError as e:
338                            logger.error(e)
339                    if len(self.packages)>1:
340                        gather_list.append(mem_device_snapshot.total_pss)
341                    if self.mem_queue:
342                        gather_list[0] = collection_time
343                        self.mem_queue.put(gather_list)
344                    if not self.mem_queue:#为了本地单个文件运行
345                        try:
346                            with open(mem_file, 'a+',encoding="utf-8") as mem_writer:
347                                writer_p = csv.writer(mem_writer, lineterminator='\n')
348                                writer_p.writerow(gather_list)
349                                logger.debug("write to file:" + mem_file)
350                                logger.debug(gather_list)
351                        except RuntimeError as e:
352                            logger.error(e)
353
354                after = time.time()
355                time_consume = after - before
356                delta_inter = self._interval - time_consume
357                logger.info("time consume for meminfos: " + str(time_consume))
358                if delta_inter > 0:
359                    time.sleep(delta_inter)
360            except:
361                logger.error("an exception hanpend in meminfo thread, reason unkown!")
362                s = traceback.format_exc()
363                logger.debug(s)
364                if self.mem_queue:
365                    self.mem_queue.task_done()
366
367        logger.debug("stop event is set or timeout")
